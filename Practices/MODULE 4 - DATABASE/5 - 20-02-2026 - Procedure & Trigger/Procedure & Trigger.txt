DELIMITER //

	CREATE PROCEDURE selectproc (INOUT myid int)

    	SELECT * FROM customers WHERE id = myid;
    END //

DELIMITER $$

	CREATE TRIGGER mytrigger3 AFTER DELETE ON subscribers FOR EACH ROW
    
    BEGIN 
    	INSERT INTO audit_subscribers set action_performed='Deleted a  subscriber' ,fname =  old.fname;
    END $$

DELIMITER ;



DELIMITER $$

	CREATE TRIGGER mytrigger2 AFTER Update ON subscribers FOR EACH ROW
    
    BEGIN 
    	INSERT INTO audit_subscribers set action_performed='Updated a  subscriber' ,fname =  old.fname;
    END $$

DELIMITER ;


DELIMITER $$

	CREATE TRIGGER mytrigger3 BEFORE INSERT ON subscribers FOR EACH ROW
    
    BEGIN 
    	INSERT INTO audit_subscribers set action_performed='Inserted a  subscriber' ,fname =  new.fname;
    END $$

DELIMITER ;
